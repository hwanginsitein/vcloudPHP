  201.     public static function create($svc, $obj)
  202.     {
  203.         if (!isset($obj))
  204.         {
  205.             return null;
  206.         }
  207.         $url = null;
  208.         if (method_exists($obj, 'get_type')) //assume it is a data object
  209.         {
  210.             $type = $obj->get_type();
  211.             if ($type)
  212.             {
  213.                 $className = self::getClassnameByType($type);
  214.                 if ($className)
  215.                 {
  216.                     return new $className($svc, $obj->get_href());
  217.                 }
  218.                 throw new VMware_VCloud_SDK_Exception("Create SDK object " .
  219.                                             "error: no class name found\n");
  220.             }
  221.         }
  222.         else if (is_string($obj))  //assume it is an URL
  223.         {
  224.             $url = $obj;
  225.         }
  226.         else if(method_exists($obj, 'get_href'))
  227.         {
  228.             $url = $obj->get_href();
  229.         }
  230.         $className = self::getClassnameByUrl($svc, $url);
  231.         if ($className)
  232.         {
  233.             return new $className($svc, $url);
  234.         }
  235.         throw new VMware_VCloud_SDK_Exception("Create SDK object error: " .
  236.                                                "no class name found\n");
  237.     }

  239.     /**
  240.      * Get a class name of an SDK object from a type attribute of a VMware
  241.      * vCloud ReferenceType or LinkType data object.
  242.      *
  243.      * @param string $type  The 'type' attribute of a vCloud data object
  244.      * @return string|null
  245.      * @access private
  246.      */
  247.     private static function getClassnameByType($type)
  248.     {
  249.         $typeMap = self::$typeMap;
  250.         $className = "VMware_VCloud_SDK_{$typeMap[$type]}";
  251.         return class_exists($className)? $className : null;
  252.     }
  253.  
  254.     /**
  255.      * Get a class name of an SDK object from a URI of a VMware
  256.      * vCloud resource entity or the entry point of resource entity.
  257.      *
  258.      * @param VMware_VCloud_SDK_Service $svc 
  259.      * @param string $url    URL of the resource entity or the entry point
  260.      * @return string|null
  261.      * @access private
  262.      */
  263.     private static function getClassnameByUrl($svc, $url)
  264.     {
  265.         $urlMap = self::$urlMap;
  266.         $baseUrl = $svc->getBaseUrl();
  267.         $rest = substr($url, strlen($baseUrl)+1);
  268.         $restArr = explode('/', $rest);
  269.         if (preg_match('/^vm-|vapp-|vappTemplate-/', end($restArr), $matches))
  270.         {
  271.             $key = substr_replace($matches[0], '', -1);
  272.         }
  273.         else
  274.         {
  275.             if (!preg_match('/^admin|extension$/', end($restArr)))
  276.             {
  277.                 array_splice($restArr, count($restArr)-1); //remove id
  278.             }
  279.             $key = implode('/', $restArr);
  280.         }
  281.         $className = "VMware_VCloud_SDK_{$urlMap[$key]}";
  282.         return class_exists($className)? $className : null;
  283.     }
  284. }
