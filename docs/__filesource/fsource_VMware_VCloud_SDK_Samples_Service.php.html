  142.         $encodeddata= base64_encode($compressedxml);
  143.  
  144.         if (false === strpos($server, 'http'))
  145.         {
  146.             $server = "https://$server";
  147.         }
  148.         $this->setConfig($config);
  149.         $this->setAPIVersion($apiVersion);
  150.         $auth = null;
  151.          if ($org == VMware_VCloud_SDK_Constants::SYSTEM_ORG)
  152.         {
  153.             $auth = VMware_VCloud_SDK_Constants::SIGN_ATTRIBUTE . " " . VMware_VCloud_SDK_Constants::TOKEN_ATTRIBUTE . "= \"" . $encodeddata . "\"";
  154.         }
  155.         else
  156.         {
