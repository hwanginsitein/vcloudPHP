  229.     public static function getContainedLinks($ctype, $rel, $obj,
  230.                                              $method='getLink')
  231.     {
  232.         if (!method_exists($obj, $method))
  233.         {

  235.         }
  236.         $links = $obj->{$method}();
  237.         if (!$links)

  240.         }
  241.         if (!$ctype && !$rel)
  242.         {
  243.             return $links; // no filtering, returns all links
  244.         }
  245.         // when specifying ctype or rel or both
  246.         $out = array();
  247.         foreach ($links as $link)
  248.         {
  249.             if (is_null($ctype) || 0 < strpos($link->get_type(), $ctype.'+xml'))
  250.             {
  251.                 if (is_null($link) || $rel == $link->get_rel())
  252.                 {
  253.                     array_push($out, $link);
  254.                 }
  255.             }
  256.         }
  257.         return $out;
  258.     }
